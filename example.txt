type(Person) {
	name: string
	age: number
	sex: string
}

type(Plant) {
	name: string
	genus: string
	species: string
}

Person(1) {
	name: John
	age: 21
	sex: male
}

Person(2) {
	name: Mary
	age: 22
	sex: female
}

Person(3) {
	name: Dave
	age: 55
	sex: male
}

Plant(1) {
	name: Tea
	genus: Camellia
	species: Chinensis
}

	name: Stewardia
	genus: Stewardia
	species: Pseudocamellia	
}

}

	sex: eq(female)
}

}
